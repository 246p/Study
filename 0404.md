    std::string targ_line = line.substr(space_idx + 1, std::string::npos);
    int score = stoi(score_str);
    dfg_node_map[targ_line] = std::make_pair(idx++, (unsigned int) score);
    if (idx >= DFG_MAP_SIZE) {

      exit(1);
